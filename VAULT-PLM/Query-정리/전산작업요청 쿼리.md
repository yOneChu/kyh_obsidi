




### 👉 기본 조회 쿼리
```sql
select A.MD$NUMBER, -- 요청번호  
       A.MD$STATUS, -- 상태  
       A.MD$DESC,  
       A.ANSWERDETAIL, --작업내용  
       (select U.MD$DESC FROM FUSER$SF U WHERE U.MD$NUMBER = A.MD$USER) AS MUSER,  
       A.MANAGER AS MANAGER,  
       A.HOGI, -- 대표호기  
       CODN(A.PRIORITY) AS 우선순위, -- 우선순위  
       CODN(A.DESIGNPART) AS 구분, -- 전기/기계  
       CODN(A.REQUESTTYPE) AS 작업구분, --작업구분  
       CODN(A.REQUESTCAUSE) AS 요청사유, -- 요청사유  
       A.REQUESTDETAIL AS 요청내용, -- 요청내용  
       CODN(A.SUBAESUITABILITY1) AS 수배자료적합성,  
       CODN(A.SUBAESUITABILITY2) AS 수배자료적합성2,  
       CODN(A.ISLIMITCONDITION), --제한조건 작성 여부  
       CODN(A.ISLAYOUTMANAUL) AS LAYOUT_MANUAL,  
       CODN(A.ISFINISHDCB) AS ISFINISHDCB, --DCB 완료 여부  
       CODN(A.ISFINISHISIR) AS ISIR초도품검사, --ISIR(초도품 검사)  
        CODN(A.ISFINISHCERTIFY) AS 인증완료여부, --인증(모델, 부품) 완료 여부  
       CODN(A.ISHANDLINGSTOCK), -- 재고 처리 여부  
       CODN(A.ISUPDATEDUTYTABLE) AS DUTY_TABLE, --DUTY TABLE 수정요청 여부  
       CODN(A.ISAPPLYSERIES) AS 시리즈현장적용여부, --시리즈 현장 적용 여부  
       CODN(A.ISTEAMSHARED) AS 유관팀공유여부, --유관팀 공유여부  
       CODN(A.COSTINFLUENCE) AS 원가영향도,  
       CODN(A.SUBSYSSUPPLYDIV) AS SubSystem공급구분,  
       A.MD$MDATE,  
     SUBSTR(A.MD$CDATE, 1, 6) AS CRE_MONTH,  
      SUBSTR(A.MD$MDATE, 1, 6) AS MOD_MONTH,  
       DATEFORMAT(A.MD$CDATE, 'YYYYMMDDHH24MISS', 'YYYY-MM-DD HH24:MI:SS') AS CRE_DATE,  
       DATEFORMAT(A.MD$MDATE, 'YYYYMMDDHH24MISS', 'YYYY-MM-DD HH24:MI:SS') AS MOD_DATE  
FROM NEWPLMDESIGNREQUEST$VF A  
--WHERE A.MD$NUMBER = '01336'  
WHERE  
 SUBSTR(A.MD$MDATE, 1, 6) > '202401'  
  AND A.MD$NUMBER = '03187'  
AND A.REQUESTTYPE = '2805876119' -- 수배로직


```
