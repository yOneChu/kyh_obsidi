




### 👉 기본 조회 쿼리
```sql
SELECT A.MD$NUMBER AS REQNO,    --요청번호  
           COD(A.ISFINISHCERTIFY), --인증완료여부  
COD(A.ISHANDLINGSTOCK), --재고 처리 여부  
COD(A.ISUPDATEDUTYTABLE), --DUTY TABLE 수정요청 여부  
COD(A.ISAPPLYSERIES), --시리즈 현장 적용 여부  
COD(A.ISTEAMSHARED), --유관팀 공유여부  
A.MD$STATUS, -- 상태  
A.MD$DESC,  
A.MD$USER,  
(select U.MD$DESC FROM FUSER$SF U WHERE U.MD$NUMBER = A.MD$USER) AS MUSER,  
A.MANAGER AS MANAGER,  
A.HOGI, -- 대표호기  
CODN(A.PRIORITY) AS 우선순위, -- 우선순위  
CODN(A.DESIGNPART) AS 구분, -- 전기/기계  
CODN(A.REQUESTCAUSE), -- 요청사유  
A.ANSWERDETAIL,
A.REQUESTDETAIL, -- 요청내용  
A.REQUESTTYPE,  
CODN(A.REQUESTTYPE) AS 작업구분, --작업구분  
COD(A.SUBAESUITABILITY1), --수배자료적합성(유관부품 포함)  
COD(A.SUBAESUITABILITY2), --수배자료적합성 (수배조건)  
COD(A.ISLIMITCONDITION) --제한조건작성여부  
,COD(A.ISLAYOUTMANAUL) -- LAYOUT MANUAL  
,COD(A.ISFINISHDCB) -- DCB 완료 여부  
,COD(A.ISFINISHISIR) --ISIR(초도품 검사)  
,COD(A.ISFINISHCERTIFY), --인증완료여부  
COD(A.ISHANDLINGSTOCK), --재고 처리 여부  
COD(A.ISUPDATEDUTYTABLE), --DUTY TABLE 수정요청 여부  
COD(A.ISAPPLYSERIES), --시리즈 현장 적용 여부  
COD(A.ISORDERNDESIGNSITE), --기 수주/설계 현장 대응 여부  
COD(A.ISTEAMSHARED), --유관팀 공유여부  
CODN(A.COSTINFLUENCE) AS 원가영향도, -- 원가영향도  
CODN(A.SUBSYSSUPPLYDIV), --SubSystem 공급 구분  
A.MD$MDATE,  
SUBSTR(A.MD$CDATE, 1, 6) AS CRE_MONTH,  
SUBSTR(A.MD$MDATE, 1, 6) AS MOD_MONTH,  
DATEFORMAT(A.MD$CDATE, 'YYYYMMDDHH24MISS', 'YYYY-MM-DD HH24:MI:SS') AS CRE_DATE,  
DATEFORMAT(A.MD$MDATE, 'YYYYMMDDHH24MISS', 'YYYY-MM-DD HH24:MI:SS') AS MOD_DATE

FROM NEWPLMDESIGNREQUEST$VF A  
--WHERE A.MD$NUMBER = '03412'  
WHERE  
 SUBSTR(A.MD$MDATE, 1, 4) > '2024'  
AND A.MD$NUMBER = '03412'  
;  
--AND A.REQUESTTYPE = '2805876119' -- 수배로직
```
