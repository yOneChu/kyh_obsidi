### ⭐테이블 생성
```
CREATE TABLE PARTDASHBOARD (  
    partALL       NVARCHAR(100)           NOT NULL,        -- 전체 파트 수  
    partActive    NVARCHAR(100)           NOT NULL,        -- 활성 상태 파트 수  
    partInactive  NVARCHAR(100)           NOT NULL,        -- 비활성 상태 파트 수 
    OLS           NVARCHAR(100), 
    batchDate     NVARCHAR(100)      NOT NULL         -- 배치 기준일  
);
```


### ⭐블럭 별 많은 자재 TOP. 20
```
 SELECT  
A.BLOCKNO_NUMBER, COUNT(A.BLOCKNO_NUMBER) AS BLOCKNO_COUNT  
FROM normalpart$vf A JOIN normalpart$id B  
ON A.vf$ouid = B.id$last  
WHERE A.NATION != '2803457356' --중국법인 제외  
AND A.PART_STATUS = '2466425004' --활성  
AND A.MD$STATUS = 'RLS' --릴리즈  
AND SUBSTR(A.BLOCKNO_NUMBER, 2,1) IN ('1','2','3')  
AND SUBSTR(A.MD$MDATE, 0,4) = '2025'  
GROUP BY A.BLOCKNO_NUMBER  
;
```


### ⭐PLM에 등록된 전체 부품 조회
```
SELECT  
    COUNT(A.md$number)  
FROM normalpart$vf A, normalpart$id B  
WHERE A.vf$ouid = B.id$last  
AND A.NATION != '2803457356' --중국법인 제외  
AND A.PART_STATUS = '2501081338' --폐기  
AND A.MD$STATUS = 'RLS' --릴리즈  
AND SUBSTR(A.BLOCKNO_NUMBER, 2,1) IN ('1','2','3')  
;
```
