### PID 별 라인 개수 쿼리 (TOP. 50)
```
SELECT  
    PID,  
    line_count  
FROM (  
    SELECT  
        h.PID,  
        COUNT(*) AS line_count  
    FROM  
        variant_d d  
    JOIN  
        variant_h h ON h.HOUID = d.HOUID  
    JOIN  
        variant_id id ON h.HOUID = id.LAST_HOUID  
    GROUP BY  
        h.PID  
    ORDER BY  
        COUNT(*) DESC  
)  
WHERE ROWNUM <= 50;
```
