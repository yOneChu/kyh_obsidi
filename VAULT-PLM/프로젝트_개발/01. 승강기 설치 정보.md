### 2. 테이블 정보
```
CREATE TABLE ElevatorInstallation (
    elevatorNo          NVARCHAR(20)  NULL,   -- 설비번호(문자 보존)
    elvtrAsignNo        NVARCHAR(50)  NULL,   -- 호기번호
    bdmgtSn             NVARCHAR(40)  NULL,       -- 건물관리 일련
    address1            NVARCHAR(255) NULL,
    address2            NVARCHAR(255) NULL,
    buldNm              NVARCHAR(200) NULL,
    buldPrposLclas      NVARCHAR(50)  NULL,
    buldPrposSclas      NVARCHAR(50)  NULL,
    companyNm           NVARCHAR(100) NULL,
    elvtrDetailForm     NVARCHAR(50)  NULL,
    elvtrDiv            NVARCHAR(50)  NULL,
    elvtrForm           NVARCHAR(50)  NULL,
    elvtrKindNm         NVARCHAR(50)  NULL,
    frstInstallationDe  NVARCHAR(50)  NULL,
    installationDe      NVARCHAR(50)  NULL,
    installationPlace   NVARCHAR(100) NULL,
    liveLoad            NVARCHAR(50)  NULL,
    ratedCap            NVARCHAR(50)  NULL,
    ratedSpeed          NVARCHAR(50)  NULL,
    shuttleFloorCnt     NVARCHAR(50)  NULL,
    sido                NVARCHAR(50)  NULL,
    sigungu             NVARCHAR(100) NULL
);
```
