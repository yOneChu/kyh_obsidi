```SQL
SELECT A.PID, A.METHOD  
FROM COSTVARIANT_H A, COSTVARIANT_ID B  
WHERE A.PID = B.PID AND A.HOUID = B.LAST_HOUID AND A.PID LIKE 'AUTO%' AND NVL(ISFLOORSPEC, 'N') = 'N'  
ORDER BY PID;  
  
  
SELECT A.PID, A.METHOD  
FROM VARIANT_H A, VARIANT_ID B  
WHERE A.PID = B.PID AND A.HOUID = B.LAST_HOUID AND A.PID LIKE 'AUTO%' AND NVL(ISFLOORSPEC, 'N') = 'N'  
ORDER BY PID
```
